
//single inheritance

#include<iostream>
using namespace std;

class B
{
	int a;
	public:
		int b;
		B(){a=10;b=20;}
		int get_a(void){return a;}
};
class D : public B
{
	int d;
	public:
		void mul(void){d=b*get_a();}
		void disp(void);
};
void D ::disp(void)
{
	cout <<  "A = "<< get_a() <<"\n";
	cout << "B = "<< b <<"\n";
	cout << "D = "<< d <<"\n";
}

int main()
{
	D d;
	d.get_a();
	d.mul();
	d.disp();

	d.b=100;
	d.mul();
	d.disp();
}


//visibility
#include <iostream>
using namespace std;
class Inh
{
    private:
        int pri_v;
    protected:
        int pro_v;
    public:
        int pub_v;
        Inh()
        {
            pri_v=10;
            pro_v=20;
            pub_v=30;
        }
        void disp()
        {
            cout << "Base class............." << endl;
            cout << "private variable = " << pri_v << endl;
            cout << "protected variable = " << pro_v << endl;
            cout << "public variable = " << pub_v << endl;
        }
};
class Dev : private Inh
{
   private:
        int pri_d;
    protected:
        int pro_d;
    public:
        int pub_d;
        Dev()
        {
            pri_d=100;
            pro_d=200;
            pub_d=300;
        }
        void disp()
        {
            cout << "derived class............." << endl;
            cout << "private variable = " << pri_d << endl;
            cout << "protected variable = " << pro_d << endl;
            cout << "public variable = " << pub_d << endl;
        }
};
int main()
{
    Inh i;
    i.disp();
    Dev d;
    d.disp();
    return 0;
}

//pointing derived class objects
#include <iostream>
using namespace std;
class B
{
	public:
		int b=10;
		void show(void) {cout <<"b="<<b<<"\n";}
};
class D : public B
{
	public:
		int d;
		void show(void) {cout <<"D dkfkdsfkdsfjskdfjb="<<b<<"\n";cout <<"d="<<d<<"\n";}
};
int main()
{
	B b;
	D d;
	B *bptr;
	cout<<"using bptr and base object\n";
	bptr->show();	//calls B's show method
	cout <<"using bptr and derived object\n";
	bptr = &d;
	bptr->b = 20;	//access d object via bptr
	//bptr->d = 100 //will not work
	bptr->show();	//calls B's show method
	cout <<"using dptr and derived object\n";
	D *dptr = &d;
	dptr->d=100;
	dptr->show();
	cout <<"using (D *)bptr\n";
	((D *)bptr) -> d=200;
	((D *)bptr) -> show();
	return 0;
}

//virtual functions
#include <iostream>
using namespace std;
class B
{
	public:
		void disp(void){cout<<"base display\n";}
		virtual void show(void){cout <<"base show\n";}
};
class D : public B
{
	public:
		void disp(void){cout<<"derived display\n";}
		void show(void){cout <<"derived show\n";}
};
int main()
{
	B b;
	D d;
	B *bptr;
	cout << "bptr points to base\n";
	bptr = &b;
	bptr->disp();	//calls base version
	bptr->show();   //calls base version
	cout << "bptr points to derived\n";
	bptr = &d;
	bptr->disp();  	//calls base version
	bptr->show();   //calls derived version
	return 0;
}

//abastract class can have constructors
#include<iostream>
using namespace std;

// An abstract class with constructor
class Base
{
protected:
int x;
public:
virtual void fun() = 0;
Base(int i) {
			x = i;
			cout<<"Constructor of base called\n";
			}
};

class Derived: public Base
{
	int y;
public:
	Derived(int i, int j):Base(i) { y = j; }
	void fun() { cout << "x = " << x << ", y = " << y<<'\n'; }
};

int main(void)
{
	Derived d(4, 5);
	d.fun();
	
//object creation using pointer of base class
	Base *ptr=new Derived(6,7);
	ptr->fun();
	return 0;
}
