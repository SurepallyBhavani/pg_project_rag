#include <iostream>
using namespace std;

class unaryopex
{
	int x,y;
	public:
		void getdata(int a,int b);
		void disp();
		void operator-();	//overload unary minus
};
void unaryopex :: getdata(int a,int b)
{
	x=a; y=b;
}
void unaryopex :: disp()
{
	cout << "x = "<<x<<"\n";
	cout << "y = "<<y<<"\n";
}
void unaryopex :: operator-()
{
	x=-x; 	y=-y;
}
int main()
{
	unaryopex u;
	u.getdata(10,20);
	u.disp();
	-u;
	u.disp();
	return 0;
}

#include <iostream>
using namespace std;

class complex
{
	float real,img; //real and imaginary parts
	public:
		complex(){ }
		complex(float r, float i) {real=r;img=i;}
		void disp(void);
		complex operator+(complex);	//overload binaryplus
};
void complex :: disp(void)
{
	cout << real << "+J"<<img<<"\n";
}
complex complex :: operator+(complex c)
{
	complex temp;
	temp.real=real+c.real;
	temp.img=img+c.img;
	return temp;
}
int main()
{
	complex c1(1.3,2.5),c2(2.1,4.2),c3;
	c3=c1+c2;
	cout << "c1=";	c1.disp();
	cout << "c2=";	c2.disp();
	cout << "c3=";	c3.disp();
}

//complex number addition using friend
#include <iostream>
using namespace std;
class complex
{
	float real,img; //real and imaginary parts
	public:
		complex(){ }
		complex(float r, float i) {real=r;img=i;}
		void disp(void);
		friend complex operator+(complex,complex);	//overload binaryplus
};
void complex :: disp(void)
{
	cout << real << "+J"<<img<<"\n";
}
complex operator+(complex a,complex b)
{
	return complex((a.real+b.real),(a.img+b.img));
}
int main()
{
	complex c1(1.3,2.5),c2(2.1,4.2),c3;
	c3=c1+c2;
	cout << "c1=";	c1.disp();
	cout << "c2=";	c2.disp();
	cout << "c3=";	c3.disp();
}

// This function displays the status of the format flags.
void showflags()
{
    ios::fmtflags f;
    long i;
    f =  cout.flags(); // get flag settings
    // check each flag
    for(i=0x4000; i; i = i >> 1)
         if(i & f) cout << "1 ";
        else cout << "0 ";
    cout << " \n";
}

#include <iostream>
#include <fstream>
#include<conio.h>
using namespace std;
class stud
{
    int rno;
    char name[30];
    float marks;
    public:
    void getdata()
    {
        cout << "enter rno :  ";
        cin>>rno;
        cin.ignore();
        cout << "enter name :  ";
        cin.getline(name,30);
        cout << "enter marks :  ";
        cin>>marks;
    }
    void dispdata()
    {
        cout<<"Name : "<<name<<endl;
        cout<<"Roll No : "<<rno<<endl;
        cout<<"Marks : "<<marks<<endl;
    }
};
int main()
{
    stud s[3];
    fstream f;
    int i;
    f.open("stud.txt",ios::out);
    cout<<"Writing...."<<endl;
    for(i=0;i<3;i++)
    {
        s[i].getdata();
        f.write((char *)&s[i],sizeof(s[i]));
    }
    f.close();
    f.open("stud.txt",ios::in);
    cout<<"Reading ......."<<endl;
    for(i=0;i<3;i++)
    {
        f.read((char *)&s[i],sizeof(s[i]));
        s[i].dispdata();
    }
    f.close();
    return 0;
}
