//multiple catch
#include <iostream>
using namespace std;

void test(int x)
{
	try
	{	
		if (x==1)	throw x;
		else if (x==0) throw 1.9;
		else if (x==2) throw 2.1f;
		cout<< "end of try block"<<endl;
	}
	catch(int i)
	{
		cout << "caught integer"<<endl;
	}
	catch(double f)
	{
		cout << "caught double"<<endl;
	}
	catch(float f)
	{
		cout << "caught float"<< endl;
	}
	cout<<"end of try catch"<<endl;

}
int main()
{
	test(1);
	test(0);
	test(2);
	test(3);	//try block does not throw any exception
	cout << "end of main";
}

//stack unwinding example
#include <iostream> 

using namespace std; 

// A sample function f1() that throws an int exception 
void f1() throw (int)
{ 
    cout<<"\n f1() Start "; 
    throw 100; 
    cout<<"\n f1() End "; 
} 

// Another sample function f2() that calls f1() 
void f2() throw (int) 
{ 
    cout<<"\n f2() Start "; 
    f1(); 
    cout<<"\n f2() End "; 
} 

// Another sample function f3() that calls f2() and handles exception thrown by f1() 
void f3() 
{ 
    cout<<"\n f3() Start "; 
    try 
    { 
	    f2(); 
    } 
    catch(int i) 
    { 
        cout<<"\n Caught Exception: "<<i; 
    } 
    cout<<"\n f3() End"; 
} 

// A driver function to demonstrate Stack Unwinding process 
int main() { 
f3(); 

getchar(); 
return 0; 
} 



#include <iostream>
#include <cstring>
using namespace std;
class MyException 
{
    public:
        char str_what[80];
        int what;
    MyException() 
    {  *str_what=0; what = 0; }
    MyException(char *s, int e)
    {
        strcpy(str_what, s);
        what = e;
    }
};
int main()
{
    int i;
    try {
            cout << "Enter a positive number: ";
            cin >> i;
            if(i<0)
                throw MyException("Not Positive", i);
        }
        catch (MyException e) 
        { // catch an error
            cout << e.str_what << ": ";
            cout << e.what << "\n";
        }
    return 0;
}
